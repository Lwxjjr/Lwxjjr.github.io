func kthLargestPerfectSubtree(root *TreeNode, k int) int {
    ans := []int{}
    var dfs func(node *TreeNode) int
    dfs = func(node *TreeNode) int {
        if node == nil {
            return 0
        }
        leftH := dfs(node.Left)
        rightH := dfs(node.Right)
        if leftH < 0 || rightH < 0 || leftH != rightH {
            return -1 // ‰∏çÂêàÊ≥ï
        }
        ans = append(ans, leftH+1)
        return leftH + 1
    }
    dfs(root)
    // ÂÆåÁæé‰∫åÂèâÊ†ëÁöÑÊï∞Èáè < k ËøîÂõû -1
    if k > len(ans) {
        return -1
    }
    slices.Sort(ans)
    return 1<<ans[len(ans)-k] - 1
}

func countWinningSequences(s string) int {
    const mod = 1_000_000_007
    n := len(s)
    
    pow2 := make([]int, (n+1)/2)
    pow2[0] = 1
    for i := 1; i < len(pow2); i++ {
        pow2[i] = pow2[i-1] * 2 % mod
    }

    mp := [...]int{'F': 0, 'W': 1, 'E': 2}
    memo := make([][][3]int, n)
    for i := range memo {
        memo[i] = make([][3]int, n*2+1)
        for j := range memo[i] {
            memo[i][j] = [3]int{-1, -1, -1}
        }
    }
    /* 
    iÔºöÁ¨¨iÂõûÂêàÔºåÂØπÊâãÂè¨Âî§ÊòØs[i]
    diffÔºöÂàÜÂ∑Æ
    banÔºöËøôÂõûÂêàÊàë‰ª¨‰∏çËÉΩÂè¨Âî§ÁöÑÁîüÁâ©
    */
    var dfs func(int, int, int) int
    dfs = func(i, diff, ban int) (res int) {
        if -diff > i {
            return
        }
        if diff > i+1 {
            return pow2[i+1]
        }
        p := &memo[i][diff+n][ban]
        if *p != -1 {
            return *p
        }
        for k := 0; k < 3; k++ { // Êûö‰∏æËøôÂõûÂêàËÉΩÂè¨Âî§ÁöÑÁîüÁâ©
            if i == n-1 || k != ban {
                score := (k - mp[s[i]] + 3) % 3
                if score == 2 {
                    score = -1
                }
                res += dfs(i-1, diff+score, k)
            }
        }
        res %= mod
        *p = res
        return
    }
    return dfs(n-1, 0, 0)
}
