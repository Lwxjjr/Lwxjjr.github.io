func countWinningSequences(s string) int {
    const mod = 1_000_000_007
    n := len(s)
    
    pow2 := make([]int, (n+1)/2)
    pow2[0] = 1
    for i := 1; i < len(pow2); i++ {
        pow2[i] = pow2[i-1] * 2 % mod
    }

    mp := [...]int{'F': 0, 'W': 1, 'E': 2}
    memo := make([][][3]int, n)
    for i := range memo {
        memo[i] = make([][3]int, n*2+1)
        for j := range memo[i] {
            memo[i][j] = [3]int{-1, -1, -1}
        }
    }
    /* 
    iï¼šç¬¬iå›åˆï¼Œå¯¹æ‰‹å¬å”¤æ˜¯s[i]

    var dfs func(int, int, int) int
    dfs = func(i, diff, ban int) (res int) {
        if -diff > i {
            return
        }
        if diff > i+1 {
            return pow2[i+1]
        }
        p := &memo[i][diff+n][ban]
        if *p != -1 {
            return *p
        }
        for k := 0; k < 3; k++ { // æšä¸¾è¿™å›åˆèƒ½å¬å”¤çš„ç”Ÿç‰©
            if i == n-1 || k != ban {

                if score == 2 {
                    score = -1
                }
                res += dfs(i-1, diff+score, k)
            }
        }
        res %= mod
        *p = res
        return
    }
    return dfs(n-1, 0, 0)
}
